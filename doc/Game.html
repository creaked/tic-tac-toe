def play
  until win?
    turn(@x.name)
    $board[@x_move.downcase.to_sym] = Rainbow("X").red
    @board.draw
    
    turn(@o.name)
    $board[@o_move.downcase.to_sym] = Rainbow("O").green
    @board.draw
  end
end
